# P80 - 图的表示转换

> Convert between graph representations
>
> 官方模块：`Problems.P80`　核心类型：`ConvertibleGraph`

## 四种表示

```haskell
import Data.List (nub)
import qualified Data.Map.Strict as Map
import qualified Data.Set as Set

type Vertex = Int
type Edge = (Vertex, Vertex)
newtype Lists = Lists ([Vertex], [Edge]) deriving (Eq, Show)
newtype Adjacency = Adjacency [(Vertex, [Vertex])] deriving (Eq, Show)
newtype Paths = Paths [[Vertex]] deriving (Eq, Show)
newtype G = G (Map.Map Vertex (Set.Set Vertex)) deriving (Eq, Show)

normalize (u,v) | u <= v = (u,v)
                | otherwise = (v,u)
```

将每种表示先转成规范的“顶点集合 + 无向边集合”，再从规范形式构造目标类型。

### 方法一：规范顶点集与边集

#### 转为规范形式

```haskell
class ConvertibleGraph g where
  canonical :: g -> (Set.Set Vertex, Set.Set Edge)

instance ConvertibleGraph Lists where
  canonical (Lists (vs,es)) =
    (Set.fromList vs, Set.fromList (map normalize es))

instance ConvertibleGraph Adjacency where
  canonical (Adjacency rows) =
    (Set.fromList (map fst rows),
     Set.fromList [normalize (u,v) | (u,ns) <- rows, v <- ns])

instance ConvertibleGraph Paths where
  canonical (Paths ps) =
    (Set.fromList (concat ps),
     Set.fromList [normalize e | path <- ps, e <- zip path (drop 1 path)])

instance ConvertibleGraph G where
  canonical (G graph) =
    (Map.keysSet graph,
     Set.fromList [normalize (u,v) | (u,ns) <- Map.toList graph
                                   , v <- Set.toList ns])
```

#### 构造目标表示

```haskell
toLists graph = Lists (Set.toList vs, Set.toList es)
  where (vs,es) = canonical graph

toAdjacency graph = Adjacency
  [(v, [other v e | e <- Set.toList es, contains v e]) | v <- Set.toList vs]
  where
    (vs,es) = canonical graph
    contains v (a,b) = v == a || v == b
    other v (a,b) = if v == a then b else a

toPaths graph = Paths
  (map (\(u,v) -> [u,v]) (Set.toList es) ++ map (:[]) isolated)
  where
    (vs,es) = canonical graph
    touched = Set.fromList [x | (u,v) <- Set.toList es, x <- [u,v]]
    isolated = Set.toList (Set.difference vs touched)

toG graph = G $ Set.foldl addEdge base es
  where
    (vs,es) = canonical graph
    base = Map.fromSet (const Set.empty) vs
    addEdge m (u,v) = Map.insertWith Set.union u (Set.singleton v)
                    $ Map.insertWith Set.union v (Set.singleton u) m
```

### 方法二：邻接 Map 作为中间表示

```haskell
class ToAdjacencyMap g where
  adjacencyMap :: g -> Map.Map Vertex (Set.Set Vertex)

instance ToAdjacencyMap Lists where
  adjacencyMap (Lists (vertices, edges)) = foldl addUndirected base edges
    where base = Map.fromList [(vertex, Set.empty) | vertex <- vertices]

instance ToAdjacencyMap Adjacency where
  adjacencyMap (Adjacency rows) = foldl addUndirected base edges
    where
      base = Map.fromList [(vertex, Set.empty) | (vertex, _) <- rows]
      edges = [(vertex, neighbor) | (vertex, neighbors) <- rows
                                  , neighbor <- neighbors]

instance ToAdjacencyMap Paths where
  adjacencyMap (Paths paths) = foldl addUndirected base edges
    where
      base = Map.fromList [(vertex, Set.empty) | vertex <- concat paths]
      edges = concatMap (\path -> zip path (drop 1 path)) paths

instance ToAdjacencyMap G where
  adjacencyMap (G graph) = graph

addUndirected :: Map.Map Vertex (Set.Set Vertex) -> Edge
              -> Map.Map Vertex (Set.Set Vertex)
addUndirected graph (u, v) =
  Map.insertWith Set.union u (Set.singleton v)
    (Map.insertWith Set.union v (Set.singleton u) graph)

toListsViaMap :: ToAdjacencyMap g => g -> Lists
toListsViaMap graph = Lists (Map.keys table, edges)
  where
    table = adjacencyMap graph
    edges = [(u, v) | (u, neighbors) <- Map.toList table
                    , v <- Set.toList neighbors, u <= v]

toAdjacencyViaMap :: ToAdjacencyMap g => g -> Adjacency
toAdjacencyViaMap graph = Adjacency
  [(vertex, Set.toList neighbors)
  | (vertex, neighbors) <- Map.toList (adjacencyMap graph)]

toPathsViaMap :: ToAdjacencyMap g => g -> Paths
toPathsViaMap graph = Paths
  (map (\(u, v) -> [u, v]) edges ++ map (:[]) isolated)
  where
    table = adjacencyMap graph
    Lists (_, edges) = toListsViaMap graph
    isolated = [vertex | (vertex, neighbors) <- Map.toList table
                       , Set.null neighbors]

toGViaMap :: ToAdjacencyMap g => g -> G
toGViaMap = G . adjacencyMap
```

第一种路线把无向边去重后作为中心；第二种路线始终维护对称邻接 Map，再从 Map 导出其他表示。邻接查询密集时第二种更自然，比较图结构或做集合运算时第一种更方便。

## 方法对比

| 中间表示 | 优势 |
|----------|------|
| `(Set Vertex, Set Edge)` | 规范比较、边去重直接 |
| `Map Vertex (Set Vertex)` | 邻居查询和图遍历直接 |

## 测试

```haskell
>>> canonical (toG (Paths [[1,2,3],[1,3],[4]])) ==
... canonical (Lists ([1,2,3,4],[(1,2),(2,3),(1,3)]))
True
>>> canonical (toGViaMap (Paths [[1,2,3],[1,3],[4]])) ==
... canonical (toG (Paths [[1,2,3],[1,3],[4]]))
True
```
